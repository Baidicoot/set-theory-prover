\begin{align*}
    \bb{M} \bnfdef&
        x
        \tag{Type Variables}
    \\\alt&
        c
        \tag{Constant Types}
    \\\alt&
        \bb{M} \to \bb{M}
        \tag{Functions}
    \\\alt&
        \prop
        \tag{Propositions}
    \\\\\bb{S} \bnfdef&
        \forall{\bar{x}}, \bb{M}
        \tag{Type Schemes}
    \\\\\bb{T} \bnfdef&
        x
        \tag{Variables}
    \\\alt&
        c
        \tag{Constants}
    \\\alt&
        M
        \tag{Metavariables}
    \\\alt&
        \lambda x, \bb{T}
        \tag{$\lambda$-Abstractions}
    \\\alt&
        \Pi_{x : \bb{M}} \bb{T}
        \tag{$\forall$-Quantifiers}
    \\\alt&
        \bb{T} \bb{T}
        \tag{Application}
    \\\alt&
        \bb{T} \Rightarrow \bb{T}
        \tag{Implication}
    \\\alt&
        \text{let $x$ = $\bb{T}$ in $\bb{T}$}
        \tag{Let-In}
    \\\\\bb{P} \bnfdef&
        a
        \tag{Axioms}
    \\\alt&
        \bb{P} \$ \bb{P}
        \tag{Modus Ponens}
    \\\alt&
        \bb{P}[x := \bb{T}]
        \tag{Universal Elimination}
    \\\alt&
        \_
        \tag{Holes}
\end{align*}

\begin{mathpar}
    \infer[Var]
        {(x, \tau) \in \Gamma}
        {\Gamma \vdash x : \tau}
    \and \infer[Const]
        {(c, \tau) \in \Gamma}
        {\Gamma \vdash c : \tau}
    \and \infer[MetaVar]
        {(M, \tau) \in \Gamma}
        {\Gamma \vdash M : \tau}
    \and \infer[Lam]
        {\Gamma, (x, \tau_0) \vdash e : \tau_1}
        {\Gamma \vdash \lambda x. e : \tau_0 \to \tau_1}
    \and \infer[Forall]
        {\Gamma, (x, \tau) \vdash e : \prop}
        {\Gamma \vdash \Pi_{x : \tau} e : \prop}
    \and \infer[App]
        {\Gamma \vdash e_0 : \tau_0 \to \tau_1
        \\\Gamma \vdash e_1 : \tau_0}
        {\Gamma \vdash e_0 e_1 : \tau_1}
    \and \infer[Imp]
        {\Gamma \vdash e_0 : \prop
        \\\Gamma \vdash e_1 : \prop}
        {\Gamma \vdash e_0 \Rightarrow e_1 : \prop}
    \and \infer[Let]
        {\Gamma \vdash e_0 : \tau_0
        \\\bar{a} = \{a \in \text{free} (\tau_0) : a \notin \text{free} (\Gamma)\}
        \\\Gamma,(x, \forall \bar{a}. \tau_0) \vdash e_1 : \tau_1}
        {\Gamma \vdash \text{let $x$ = $e_0$ in $e_1$} : e_1}
\end{mathpar}

\begin{mathpar}
    \infer[Axiom]
        {(a, e) \in \Delta}
        {\Delta \vdash a : e}
    \and \infer[ModPon]
        {\Delta \vdash p_0 : e_0 \Rightarrow e_1
        \\\Delta \vdash p_1 : e_0}
        {\Delta \vdash p_0 \$ p_1 : e_1}
    \and \infer[UniElim]
        {\Gamma \vdash e_0 : \tau
        \\\Delta \vdash p : \Pi_{x : \tau} e_1}
        {\Delta \vdash p[x := e_0] : e_1}
    \and \infer[Hole]
        { }
        {\Delta \vdash \_ : e}
\end{mathpar}
